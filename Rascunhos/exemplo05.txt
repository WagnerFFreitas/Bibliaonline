5. API (Exemplo em PHP):

Crie um arquivo chamado "api/versiculos.php" com o seguinte código:

<?php
header('Content-Type: application/json');

$conn = mysqli_connect("localhost", "usuario", "senha", "biblia");

if (!$conn) {
    die("Connection failed: " . mysqli_connect_error());
}

$versiculoId = $_GET["id"];

$sql = "SELECT * FROM versiculos WHERE id = $versiculoId";
$result = mysqli_query($conn, $sql);

if (mysqli_num_rows($result) > 0) {
    $row = mysqli_fetch_assoc($result);
    echo json_encode($row);
} else {
    echo json_encode(array("error" => "Versículo não encontrado"));
}

mysqli_close($conn);
?>
