// Preencher a lista de livros
const listaLivros = document.getElementById('livros');
livros.forEach(livro => {
    const li = document.createElement('li');
    li.textContent = livro;
    listaLivros.appendChild(li);
});

// Preencher o select de livros
const livroSelect = document.getElementById('livro-select');
livros.forEach(livro => {
    const option = document.createElement('option');
    option.value = livro;
    option.textContent = livro;
    livroSelect.appendChild(option);
});
